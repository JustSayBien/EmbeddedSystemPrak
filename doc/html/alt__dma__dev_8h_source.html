00051 /*
00052  * This header contains the device driver interface for accessing DMA 
00053  * resources. See alt_dma.h for the DMA application side interface.
00054  *
00055  * The interface model treats a DMA transaction as being composed of two 
00056  * halves (read and write). 
00057  *
00058  * An "alt_dma_txchan_dev" is used to describe the device associated with a
00059  * DMA transmit channel. An "alt_dma_rxchan_dev" is used to describe the
00060  * device associated with a DMA receive channel. 
00061  */

00063 /*
00064  * List of generic ioctl requests that may be supported by a DMA device.
00065  *
00066  * ALT_DMA_RX_ONLY_ON:    This causes a DMA channel to operate in a mode
00067  *                        where only the receiver is under software control.
00068  *                        The other side reads continously from a single
00069  *                        location. The address to read is the argument to
00070  *                        this request.
00071  * ALT_DMA_RX_ONLY_OFF:   Return to the default mode where both the receive
00072  *                        and transmit sides of the DMA can be under software
00073  *                        control.
00074  * ALT_DMA_TX_ONLY_ON:    This causes a DMA channel to operate in a mode
00075  *                        where only the transmitter is under software control.
00076  *                        The other side writes continously to a single
00077  *                        location. The address to write to is the argument to
00078  *                        this request.
00079  * ALT_DMA_TX_ONLY_OFF:   Return to the default mode where both the receive
00080  *                        and transmit sides of the DMA can be under software
00081  *                        control.
00082  * ALT_DMA_SET_MODE_8:    Transfer data in units of 8 bits.
00083  * ALT_DMA_SET_MODE_16:   Transfer data in units of 16 bits.
00084  * ALT_DMA_SET_MODE_32:   Transfer data in units of 32 bits.
00085  * ALT_DMA_SET_MODE_64:   Transfer data in units of 64 bits.
00086  * ALT_DMA_SET_MODE_128:  Transfer data in units of 128 bits.
00087  * ALT_DMA_GET_MODE:      Get the current transfer mode.
00088  *
00089  * The use of the macros: ALT_DMA_TX_STREAM_ON, ALT_DMA_TX_STREAM_OFF
00090  * ALT_DMA_RX_STREAM_OFF and ALT_DMA_RX_STREAM_ON are depreciated. You should
00091  * instead use the macros: ALT_DMA_RX_ONLY_ON, ALT_DMA_RX_ONLY_OFF, 
00092  * ALT_DMA_TX_ONLY_ON and ALT_DMA_TX_ONLY_OFF.
00093  */
